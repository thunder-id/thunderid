    {icon: <VueLogo size={64} />, title: 'Vue SDK', packageName: '@thunderid/vue', packageManager: 'npm', description: `Add authentication to your Vue.js application with ${productName} SDK.`, comingSoon: true, category: 'frontend'},

    {icon: <FlutterLogo size={64} />, title: 'Flutter SDK', packageName: 'asgardeo_flutter', packageManager: 'pub', description: `Cross-platform Flutter SDK for ${productName} authentication.`, comingSoon: true, category: 'mobile'},

    {key: 'all', label: 'All', icon: <svg width="16" height="16" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round"><rect x="3" y="3" width="7" height="7"/><rect x="14" y="3" width="7" height="7"/><rect x="14" y="14" width="7" height="7"/><rect x="3" y="14" width="7" height="7"/></svg>},

    const matchesCategory = activeFilter === 'all' || sdk.category === activeFilter;

          sx={{

          <Typography

              lineHeight: 1.15,
              textAlign: 'center',
            }}
          >

          </Typography>

            variant="body1"
            color="text.secondary"

          <Typography variant="h6" color="text.secondary">
